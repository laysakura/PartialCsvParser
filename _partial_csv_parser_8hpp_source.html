140 inline void _get_current_line(
141  const char * const text,
142  size_t text_length_byte,
143  size_t current_pos,
144  char line_terminator,
145  /* out */
146  const char ** line,
147  size_t * line_length_byte)
148 {
149  ASSERT(text);
150  ASSERT(text_length_byte >= 1);
151  ASSERT(0 <= current_pos);
152  ASSERT(current_pos <= text_length_byte - 1);
153 
154  // \n aaaaaaaaaaaa \n
155  // ^ ^
156  // start end
157  const char *line_start = text + current_pos, *line_end = text + current_pos;
158 
159  // search line_start
160  while (line_start > text && *(line_start - 1) != line_terminator) --line_start;
161  // search line_end
162  while (line_end < text + text_length_byte && *line_end != line_terminator) ++line_end;
163 
164  *line = line_start;
165  *line_length_byte = line_end - line_start;
166 }

168 inline std::vector<std::string> _split(const char * const str, size_t len, char delimiter) {
169  ASSERT(str);
170  ASSERT(len >= 0);
171 
172  std::vector<std::string> ret; // NRVO optimization may prevent copy when returning this local variable.
173 
174  const char *p_beg = str, *p_end = str;
175  while (p_end - str < len) {
176  // come to delimiter
177  if (*p_end == delimiter) {
178  ret.push_back(std::string(p_beg, p_end - p_beg));
179  p_beg = p_end + 1;
180  }
181  ++p_end;
182  }
183  // come to end of str
184  ret.push_back(std::string(p_beg, p_end - p_beg));
185  return ret;

365  inline std::vector<std::string> get_row() throw(PCPCsvError) {
366  while (cur_pos <= parse_to) {
367  const char * line;
368  size_t line_length;
369  _get_current_line(csv_config.content(), csv_config.filesize(), cur_pos, csv_config.get_line_terminator(), &line, &line_length);
370 
371  // cur_pos exactly is the beginning of current line.
372  //
373  // (\n or beginning of CSV file) aaaaaaaaaaaaaa \n
374  // <----...
375  // cur_pos
376  //
377  // Parse "aaaaaaaaaaaaaa" and move cur_pos to the beginning of the next line.
378  if (csv_config.content() + cur_pos == line) {
379  cur_pos += line_length + 1; // +1 is from line_delimitor
380 
381  const std::vector<std::string> & columns = _split(line, line_length, csv_config.get_field_terminator());
382  if (columns.size() != csv_config.get_n_columns()) {
383  std::ostringstream ss; \
384  ss << "The following line has " << columns.size() << " columns, while the first line has " << csv_config.get_n_columns() << " columns." << std::endl << std::string(line, line_length);
385  throw PCPCsvError(ss.str());
386  }
387  return columns;
388  }
389 
390  // parse_to is at the same line with cur_pos.
391  //
392  // (\n or beginning of CSV file) aaaaaaaaaaaaaa \n
393  // <---------->
394  // cur_pos parse_to

396  return std::vector<std::string>(0);
